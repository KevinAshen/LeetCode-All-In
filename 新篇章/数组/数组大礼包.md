class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
        int size = numbers.size();
        if (size == 0 || size == 1) {
            return {};
        }
        int left = 0, right = size - 1;
        int sum = numbers[left] + numbers[right];
        while (sum != target) {
            while (sum < target) {
                left++;
                sum = numbers[left] + numbers[right];
            }
            while (sum > target) {
                right--;
                sum = numbers[left] + numbers[right];
            }
        }
        return {left + 1, right + 1};
    }
};
